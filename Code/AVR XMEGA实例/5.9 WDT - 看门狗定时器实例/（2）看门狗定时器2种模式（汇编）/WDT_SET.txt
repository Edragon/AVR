;WATCH-DOG   WDT 普通模式 4S
WATCH_DOG:
	   	LDI	R16,0X27	//普通看门狗模式，超时周期4s
		LDI R17,CCP_IOREG_gc
  		STS CPU_CCP,R17
		STS WDT_CTRL,R16
WAIT_WDT_SET_OK:
    	LDS R16,WDT_STATUS
		SBRC R16,WDT_SYNCBUSY_bp//SET OK
		JMP WAIT_WDT_SET_OK

		LDS R16,WDT_CTRL
		SBRS R16,WDT_ENABLE_BP//WDT ENABLE
		JMP  WATCH_DOG
